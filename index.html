<script>

window.print();
    let count = 0;
    const tableBody = document.getElementById('tableBody');

    function addRow() {
        count++;
        const tr = document.createElement('tr');
        tr.innerHTML = `
            <td data-label="No.">${count}.</td>
            <td data-label="Tanggal"><input type="date"></td>
            <td data-label="Judul"><input type="text" placeholder="..."></td>
            <td data-label="Like"><input type="file" accept="image/*" onchange="pv(this, 'l${count}')"><div class="preview-box"><img id="l${count}"></div></td>
            <td data-label="Com"><input type="file" accept="image/*" onchange="pv(this, 'c${count}')"><div class="preview-box"><img id="c${count}"></div></td>
            <td data-label="Share"><input type="file" accept="image/*" onchange="pv(this, 's${count}')"><div class="preview-box"><img id="s${count}"></div></td>
        `;
        tableBody.appendChild(tr);
    }

    function removeRow() {
        if (count > 2) { tableBody.removeChild(tableBody.lastChild); count--; }
        else { alert("Minimal harus ada 2 baris laporan."); }
    }

    function pv(input, id) {
        const file = input.files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = (e) => {
                const img = document.getElementById(id);
                img.src = e.target.result;
                img.style.display = 'block';
            };
            reader.readAsDataURL(file);
        }
    }

    document.addEventListener('DOMContentLoaded', () => {
        addRow(); addRow(); 
        
    });

</script>
